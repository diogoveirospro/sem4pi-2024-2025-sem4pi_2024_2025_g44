DSL version 1.1.23;

DroneType droneTypeY;

Position aPos = (0,0,0);
Position anotherPos = (0, 15, 0);
Position zAxis = (0,0,1);

Velocity aVelocity = 6;
Velocity rotVelocity = PI/8;

Distance aLength = 25;

Line aLine(aPos, aLength, droneTypeY);

before
   aLine.lightsOn(YELLOW);
   aLine.move((0, 0, 1),50, aVelocity);
endbefore

aLine.rotate(aPos, zAxis, -2*PI, rotVelocity);

pause(10);

aLine.lightsOn(GREEN);

after
   aLine.move((0, 0, -1),50, aVelocity);
   aLine.lightsOff;
endafter
